ON PYTHON DEFAULT MUTABLE ARGS example:
def append_to(item, lst: list = []):
    lst.append(item)
    return lst

print(append_to(1))  # [1]
print(append_to(2))  # [1, 2] when we should expect [2]
print(append_to(3))  # [1, 2, 3] when we should expect [3]

N = 1 = 0001  → bit 0
E = 2 = 0010  → bit 1
S = 4 = 0100  → bit 2
W = 8 = 1000  → bit 3
